public class Lancero extends Soldado {
    private int longitudLanza;

    public Lancero(String nombre, int nivelVida, int fila, int columna, int longitudLanza) {
        super(nombre, 5, 10, nivelVida, fila, columna);
        this.longitudLanza = longitudLanza;
    }

    @Override
    public void accionEspecial() {
        System.out.println(getNombre() + " forma un schiltrom, aumentando su defensa en 2 puntos.");
    }

    @Override
    public String toString() {
        return super.toString() + " (Lancero)";
    }
}
