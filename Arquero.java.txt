public class Arquero extends Soldado {
    private int flechasDisponibles;

    public Arquero(String nombre, int nivelVida, int fila, int columna, int flechasDisponibles) {
        super(nombre, 7, 3, nivelVida, fila, columna);
        this.flechasDisponibles = flechasDisponibles;
    }

    public void disparar() {
        if (flechasDisponibles  0) {
            flechasDisponibles--;
            System.out.println(getNombre() +  dispara una flecha. Flechas restantes  + flechasDisponibles);
        } else {
            System.out.println(getNombre() +  no tiene flechas.);
        }
    }

    @Override
    public void accionEspecial() {
        disparar();
    }

    @Override
    public String toString() {
        return super.toString() +  (Arquero);
    }
}
