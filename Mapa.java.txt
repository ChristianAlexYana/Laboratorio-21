import java.util.ArrayList;
import java.util.Random;

public class Mapa {
    private String tipoTerritorio;
    private Soldado[][] tablero;
    private ArrayList<Soldado> ejercito1;
    private ArrayList<Soldado> ejercito2;

    public Mapa(String tipoTerritorio, int filas, int columnas) {
        this.tipoTerritorio = tipoTerritorio;
        this.tablero = new Soldado[filas][columnas];
        this.ejercito1 = new ArrayList<>();
        this.ejercito2 = new ArrayList<>();
    }

    public void agregarSoldado(Soldado soldado, int ejercito) {
        if (tablero[soldado.getFila()][soldado.getColumna()] == null) {
            if (ejercito == 1) {
                ejercito1.add(soldado);
            } else {
                ejercito2.add(soldado);
            }
            tablero[soldado.getFila()][soldado.getColumna()] = soldado;
        }
    }

    public void mostrarMapa() {
        for (int i = 0; i < tablero.length; i++) {
            for (int j = 0; j < tablero[i].length; j++) {
                if (tablero[i][j] != null) {
                    System.out.print(tablero[i][j].getNombre().charAt(0) + " ");
                } else {
                    System.out.print(". ");
                }
            }
            System.out.println();
        }
    }

    public void calcularEstadisticas() {
        System.out.println("Estadísticas del Ejército 1:");
        calcularEstadisticasEjercito(ejercito1);
        System.out.println("Estadísticas del Ejército 2:");
        calcularEstadisticasEjercito(ejercito2);
    }

    private void calcularEstadisticasEjercito(ArrayList<Soldado> ejercito) {
        int vidaTotal = 0;
        for (Soldado s : ejercito) {
            vidaTotal += s.getNivelVida();
        }
        System.out.println("Vida Total: " + vidaTotal);
        System.out.println("Promedio de Vida: " + vidaTotal / (double) ejercito.size());
    }
}
